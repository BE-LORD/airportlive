## VIDEO IMPLEMENTATION IN NEXT.JS
## ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

```typescript
// components/ui/VideoLoop.tsx — Reusable loop video component
interface VideoLoopProps {
  src: string           // /videos/hero-loop.mp4
  poster: string        // /images/hero-poster.webp (static frame)
  opacity?: number      // 0.6 default
  className?: string
  mobileHide?: boolean  // hide on mobile if true
}

export default function VideoLoop({
  src, poster, opacity = 0.6, className = '', mobileHide = false
}: VideoLoopProps) {
  return (
    <video
      autoPlay
      muted
      loop
      playsInline
      poster={poster}
      style={{ opacity, objectFit: 'cover' }}
      className={`absolute inset-0 w-full h-full ${mobileHide ? 'hidden md:block' : ''} ${className}`}
    >
      <source src={src} type="video/mp4" />
    </video>
  )
}

// Usage in Hero.tsx:
// <VideoLoop src="/videos/hero-loop.mp4" poster="/images/hero-poster.webp" opacity={0.5} mobileHide />
```

## VIDEO OPTIMIZATION COMMANDS
## ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

```bash
# Install FFmpeg (if not installed)
brew install ffmpeg  # Mac
# or: winget install ffmpeg  # Windows

# ════════════════════════════════════════
# Compress hero video to web-safe size
# ════════════════════════════════════════
ffmpeg -i hero-4k-original.mp4 \
  -vcodec libx264 \
  -crf 23 \
  -preset slow \
  -vf "scale=1920:-2" \
  -movflags faststart \
  -an \
  hero-loop.mp4

# Target: Under 8MB. If still too large, increase CRF to 26

# ════════════════════════════════════════
# Extract poster frame (first frame as static image)
# ════════════════════════════════════════
ffmpeg -i hero-loop.mp4 -vframes 1 -q:v 2 hero-poster.jpg
convert hero-poster.jpg hero-poster.webp  # Convert to WebP

# ════════════════════════════════════════
# Compress all videos in batch
# ════════════════════════════════════════
for f in *.mp4; do
  ffmpeg -i "$f" -vcodec libx264 -crf 24 -preset slow \
    -vf "scale=1920:-2" -movflags faststart -an \
    "compressed_${f}"
done

echo "All videos compressed. Check sizes with: ls -lh *.mp4"
```
